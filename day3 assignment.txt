

Join

1. Display department name and manager first name.

		select dept.department_name, e.first_name 
	from departments dept inner join employees e on dept.manager_id = e.employee_id;

2. Display department name,manager name and city

	select dept.department_name , loc.city , e.first_name from departments dept ,
locations loc,employees e where dept.location_id = loc.location_id and dept.manager_id = e.employee_id(+)

3. Display country name,city and department name.

				select dept.department_name, 
	loc.city ,loc.country_id from departments dept,locations loc where dept.location_id = loc.location_id ;


Oracle function with group by

1. Display first name, salary and round the salary to thousands.

	select first_name,salary,round(salary,-4) from employees;
		
2. Display first name and experience of the employees

	select first_name,round(months_between(sysdate,hire_date)/12) as yearsofexperience from employees;

3. Display employee details where year of experience is greater than 15 years

	select first_name,round(months_between(sysdate,hire_date)/12) as yearsofexperience 
	from employees where round(months_between(sysdate,hire_date)/12)>15;	

4. Display first name of employee who joined in 1995

	select first_name,to_char(hire_date,'yyyy') as joindate from employees where to_char(hire_date,'yyyy')=1995;

5. Display manager id and number of employees managed by the manager.
	
	select manager_id,count(employee_id) as nofemp from employees  group by manager_id;

6. Display the month in which more than 5 employees joined in any department.

	select substr(hire_date,3,4),count(employee_id)
	 from employees group by substr(hire_date,3,4) having count(employee_id)>5;   		

7. Display details of department in which the maximum salary is more than 10000

	select d1.department_id from departments d1,employees e1 where e1.department_id=d1.department_id
		 group by d1.department_id having max(salary)>10000;

8. Display department name,manger name and salary of the manger for all managers where
	experience is more than 5 years.

	
	select d1.department_name,e1.first_name,e1.salary from departments d1 inner join employees e1 on 
	e1.department_id=d1.department_id where round(months_between(sysdate,hire_date)/12)>25;
